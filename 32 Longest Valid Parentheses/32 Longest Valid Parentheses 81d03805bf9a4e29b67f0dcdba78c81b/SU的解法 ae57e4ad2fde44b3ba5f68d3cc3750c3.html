int longestValidParentheses(char * s)
{
	int left = 0;
	int right = 0;
	int len = 0;
	int index = 0;
	int i;
	int maxlen=0;
	for(i=0;i<strlen(s);i++)//left to right
	{
		if(s[i] == '(')
		{
			left++;
		}
		else
		{
			right++;
		}
		if(left == right)
		{
			len = left*2;
		}
		if(left<right||i == strlen(s) - 1)
		{
			right = 0;
			left = 0;
			if(len>maxlen)
			{
				maxlen = len;
			}
			len = 0;
		}
	}
	for(i=strlen(s)-1;i>=0;i--)//right to left
	{
		if(s[i] == '(')
		{
			left++;
		}
		else
		{
			right++;
		}
		if(left == right)
		{
			len = left*2;
		}
		if(left>right||i == 0)
		{
			right = 0;
			left = 0;
			if(len>maxlen)
			{
				maxlen = len;
			}
			len = 0;
		}
	}
	return maxlen;
}

解題說明

從左邊往右邊數左右括號個數

當遇到left == right

結算長度len = left*2

當遇到left<right||i==s[i] -1⇒代表遇到不符合字串或到結尾時結算

從右邊往左邊數左右括號

當遇到left == right

結算長度len = left*2

當遇到left>right||i==s[i] -1⇒代表遇到不符合字串或到結尾時結算

每次結算時和maxlen做比較 比較大的留下來

return maxlen

⏳ 時間複雜度：O(2N)

💾 空間複雜度：O(1)
