int longestValidParentheses(char * s)
{
	int left = 0;
	int right = 0;
	int len = 0;
	int index = 0;
	int i;
	int maxlen=0;
	for(i=0;i<strlen(s);i++)//left to right
	{
		if(s[i] == '(')
		{
			left++;
		}
		else
		{
			right++;
		}
		if(left == right)
		{
			len = left*2;
		}
		if(left<right||i == strlen(s) - 1)
		{
			right = 0;
			left = 0;
			if(len>maxlen)
			{
				maxlen = len;
			}
			len = 0;
		}
	}
	for(i=strlen(s)-1;i>=0;i--)//right to left
	{
		if(s[i] == '(')
		{
			left++;
		}
		else
		{
			right++;
		}
		if(left == right)
		{
			len = left*2;
		}
		if(left>right||i == 0)
		{
			right = 0;
			left = 0;
			if(len>maxlen)
			{
				maxlen = len;
			}
			len = 0;
		}
	}
	return maxlen;
}
