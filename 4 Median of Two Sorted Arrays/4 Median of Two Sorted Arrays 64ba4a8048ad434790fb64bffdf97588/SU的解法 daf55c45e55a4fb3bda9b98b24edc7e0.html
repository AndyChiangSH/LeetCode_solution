double findMedianSortedArrays(int* nums1, int nums1Size, int* nums2, int nums2Size)
{
int total,half;
total = nums1Size + nums2Size;
half = total / 2;
if(nums1Size > nums2Size)
{
int *temp;
temp = nums1;
nums1 = nums2;
nums2 = temp;
int temp2;
temp2 = nums1Size;
nums1Size = nums2Size;
nums2Size = temp2;
}
int l = 0;
int r = nums1Size - 1;
int i,j,num1left,num1right,num2left,num2right;
while(1)
{
i = floor(l + (r - l) / 2.0);//num1 index
j = half - i - 2;//num2 index
if(i>=0)
{
num1left = nums1[i];
}
else
{
num1left = INT_MIN;
}
if((i+1)<nums1Size)
{
num1right = nums1[i+1];
}
else
{
num1right = INT_MAX;
}
if(j>=0)
{
num2left = nums2[j];
}
else
{
num2left = INT_MIN;
}
if((j+1)<nums2Size)
{
num2right = nums2[j+1];
}
else
{
num2right = INT_MAX;
}
if(num1left<=num2right && num2left<=num1right)
{
//odd
if(total % 2)
{
return findmin(num1right,num2right);
}
//even
else
{
return (findmax(num1left,num2left) + findmin(num1right,num2right))/2.0;
}
}
else if(num1left>num2right)
{
r = i-1;
}
else//num2left>num1right
{
l=i+1;
}
}
}
int findmin(int A,int B)
{
if(A<=B)
{
return A;
}
else
{
return B;
}
}
int findmax(int A,int B)
{
if(A>=B)
{
return A;
}
else
{
return B;
}
}

解題說明

先將陣列長度較小的令為陣列A,長度較大的令為陣列B

將整個陣列合起來看 找中位數就是要將陣列分成左右兩半 左邊個數等於右邊

利用二分搜尋的方法將陣列A分成兩半(i= floor(l+(r-l)/2.0)) 左半邊的最右邊的數稱為Aleft 右半邊最左邊的數稱為Aright

將B陣列的切點切在j = half - i -2(因為陣列從0開始) 左半邊 左半邊的最右邊的數稱為Bleft 右半邊最左邊的數稱為Bright

如果超出左邊邊界設為-infinite 如果超出右邊邊界設為+infinite

接下來判斷這個切割方式有沒有符合正確切割

if(Aleft≤Bright && Bleft ≤ Aright)//正確切割

分成兩種情況 總陣列各數為

1.奇數//A和B右邊找最小

findmin(Aright,Bright);

2.偶數//A和B左邊找最大 右邊找最小 相加除以2

(findmax(Aleft,Bleft) + findmin(Aright,Bright))/2.0

else if(Aleft>Bright)//A向左找

r = i-1;

else //if(Bleft>Aright) A向右找

l = i+1;
