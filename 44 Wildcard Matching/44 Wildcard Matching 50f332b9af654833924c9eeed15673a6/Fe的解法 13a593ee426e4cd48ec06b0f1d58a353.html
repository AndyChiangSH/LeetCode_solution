                return False
        if p[-1] != '*':
            if re.search(sub_p[-1]+'$', s) is None:

            outcome = re.search(unit, s)
            if outcome is None:
                return False
            else:
                s = s[s.index(outcome[0]) + len(outcome[0]):]

        return True

        P = re.sub('(\*+)',"*",P)

        detect = re.compile(r'%s'%P)

            return True
        else:
            return False
