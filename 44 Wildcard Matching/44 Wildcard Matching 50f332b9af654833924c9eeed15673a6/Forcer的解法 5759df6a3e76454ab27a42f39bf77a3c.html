    bool isMatch(char* s,char *p)
{
	char *last=s;//save last match position
	char *pstar=NULL;
	
	while(*s)
	{
		if(*s==*p || *p=='?')
		{
			s++;
			p++;
		}
		else if(*p=='*')
		{
			//update last one
			pstar=p++;
			last=s;
		}
		else if(pstar)
		{
			p=pstar+1;	
			s=++last;	//back to last * position
		}
		else
			return false;
	}
	while(*p!='\0')//check remaing
	if(*p=='*')	p++;
	else return false;return true;
}
