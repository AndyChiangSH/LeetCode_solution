    bool isMatch(char * s, char * p){
	char *ss = s,*sp = NULL;
	while(*s) {
		if(*s == *p || *p == '?') {
			s++;
			p++;
			continue;
		}
		if(*p == '*') {
			sp = ++p;
			ss = s;
			continue;
		}
		if(sp) {
			s = ++ss;
			p = sp;
			continue;
		}
		return false;
	}
	while(*p=='*') p++;
	return !*p;
}
