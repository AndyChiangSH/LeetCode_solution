    class Solution {
    public boolean isMatch(String s, String p) {
       // p = p.replaceAll("(\\*{2,})", "*");
        int p_point=0,s_point=0;
        int p_len=p.length();
        int s_len=s.length();
        int left=p_len;
        int i;
        boolean is_star=false;
        if(p_len>0){
            for(i=0;;i++){
                if(i>=(p_len-1)){
                    if(p.charAt(i)=='*'){
                        left--;
                    }
                    is_star=true;
                    break;
                }
                if(p.charAt(i)=='*'){
                    left--;
                }
            }
        }
        int star_at=-1,before_eaten=-1,last_l=0;
        while(s_point<s_len){
            if(left>(s_len-s_point)) {return false;}
            if(p_point<p_len && s.charAt(s_point)==p.charAt(p_point)){
                s_point++;
                p_point++;
                left--;
            }
            else if(p_point<p_len && p.charAt(p_point)=='?'){
                s_point++;
                p_point++;
                left--;
            }
            else if(p_point<p_len && p.charAt(p_point)=='*'){
                star_at=p_point;
                before_eaten=s_point;
                p_point++;
                last_l=left;
            }
            else if(star_at!=-1){
                p_point=star_at+1;
                before_eaten++;
                s_point=before_eaten;
                left=last_l;
            }
            else return false;
        }
        if(left==0 && (is_star || s_point>=s_len)) return true;
        else return false;
    }
}

解題說明

left : p最少需要的坑(p-'*')

is_star : p最後是否是*

本來以為替換掉重複*會比較快，結果p = p.replaceAll("(\\*{2,})", "*");本身就超慢

⏳ 時間複雜度：O(N×M)

💾 空間複雜度：O(1)
