var nearestPalindromic = function(n) {
    'use strict';
    
    return nearestPalindromic(n);
    
    function nearestPalindromic(n) {
        let len = n.length;
        if(len === 1) {
            return String(parseInt(n, 10) - 1);
        }
        else if(len === 2) {
            let num = parseInt(n, 10);
            if(num <= 11) {
                return '9';
            } 
            else if(num <= 16) {
                return '11';
            }
            else if(num <= 19) {
                return '22';
            }
        }

        let helf = Math.floor(len / 2);
        let [ceil, floor] = [[], []];

        let index;
        for(index = 0; index < helf; index++) {
            ceil.push(n.charAt(index));
            floor.push(n.charAt(index));
        }
        let eventFlag = 0;
        for(index -= 1; index >= 0; index--) {
            if(n.charAt(index) > n.charAt(len - index - 1)) {
                eventFlag = 1;
                break;
            }
            else if(n.charAt(index) < n.charAt(len - index - 1)) {
                eventFlag = -1;
                break;
            }
        }

        if(len % 2 !== 0) {
            ceil.push(n.charAt(helf));
            floor.push(n.charAt(helf));
        }
        let rightHelf = ceil.length;

        let ceilGap, floorGap;
        let carry, start, eventRightHelf;
        if(eventFlag > 0) {
            [ceil, ceilGap] = palindromeHelf(ceil, helf);
        }
        else {
            [ceil, carry, eventRightHelf, start] = carryProcess(ceil, (+1), len);
            if(carry) { start -= 1; }
            for(let i = start; i >= 0; i--) {
                ceil.push(ceil[i]);
            }
            ceil = ceil.join('');
            ceilGap = parseInt('1' + ceil.substring(eventRightHelf)) - parseInt(n.substring(rightHelf));
        }

        if(eventFlag < 0) {
            [floor, floorGap] = palindromeHelf(floor, helf);
        }
        else {
            [floor, carry, eventRightHelf, start] = carryProcess(floor, (-1), len);
            for(let i = start; i >= 0; i--) {
                floor.push(floor[i]);
            }
            if(carry) { floor.push('9'); }
            floor = floor.join('');
            floorGap = parseInt('1' + n.substring(rightHelf)) - parseInt(floor.substring(eventRightHelf));
        }

        return (floorGap <= ceilGap)?floor:ceil;   
    }
    
    
    function palindromeHelf(event, n_helf, rightHelf = event.length) {
        for(let i = n_helf - 1; i >= 0; i--) {
            event.push(n.charAt(i));
        }
        event = event.join('');
        return [event, Math.abs(parseInt(event.substring(rightHelf)) - parseInt(n.substring(rightHelf)))];
    }

    function carryProcess(event, adjust, n_len) {
        event = parseInt(event.join(''));
        let carry = Math.floor(Math.log10(event));
        event += adjust;
        carry = (carry !== Math.floor(Math.log10(event)));
        event = String(event).split('');
        return [event, carry, event.length, (n_len % 2 !== 0)?event.length-2:event.length-1];
    }
};
